17  package org.mal_lang.langspec.step;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import jakarta.json.JsonObject;
22  import java.util.Map;
23  import org.mal_lang.langspec.Asset;
24  import org.mal_lang.langspec.Variable;
25  import org.mal_lang.langspec.builders.step.StepDifferenceBuilder;
26  
27  /**
28   * Immutable class representing a difference step in a MAL language.
29   *
30   * @since 1.0.0
31   */
32  public final class StepDifference extends StepBinaryOperation {
33    private StepDifference(
34        Asset sourceAsset, Asset targetAsset, StepExpression lhs, StepExpression rhs) {
35      super(sourceAsset, targetAsset, lhs, rhs);
36    }
37  
38    @Override
39    public JsonObject toJson() {
40      return this.toJson("difference");
41    }
42  
43    static StepDifference fromBuilder(
44        StepDifferenceBuilder builder,
45        Asset sourceAsset,
46        Map<String, Asset> assets,
47        Map<Variable, Asset> variableTargets) {
48      requireNonNull(builder);
49      requireNonNull(sourceAsset);
50      requireNonNull(assets);
51      requireNonNull(variableTargets);
52      var lhs = StepExpression.fromBuilder(builder.getLhs(), sourceAsset, assets, variableTargets);
53      var rhs = StepExpression.fromBuilder(builder.getRhs(), sourceAsset, assets, variableTargets);
54      var targetAsset = Asset.leastUpperBound(lhs.getTargetAsset(), rhs.getTargetAsset());
55      return new StepDifference(sourceAsset, targetAsset, lhs, rhs);
56    }
57  }
