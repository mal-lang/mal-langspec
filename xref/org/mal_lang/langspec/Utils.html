33  /** Utility class containing static helper functions. */
34  public final class Utils {
35    private static final String LANG_SCHEMA_JSON = "lang.schema.json";
36    private static final JsonValidationService JSON_VALIDATION_SERVICE =
37        JsonValidationService.newInstance();
38  
39    private Utils() {}

41    /**
42     * Checks that {@code objs} and all object in {@code objs} are not {@code null}.
43     *
44     * @param objs the objects to check
45     * @throws NullPointerException if {@code objs} or any object in {@code objs} is {@code null}
46     */
47    public static void checkNotNull(Object... objs) {
48      if (objs == null) {
49        throw new NullPointerException();
50      }
51      for (var obj : objs) {
52        if (obj == null) {
53          throw new NullPointerException();
54        }
55      }
56    }
57  
58    /**
59     * Checks that {@code objs} and all object in {@code objs} are not {@code null}.
60     *
61     * @param objs the objects to check
62     * @throws NullPointerException if {@code objs} or any object in {@code objs} is {@code null}
63     */
64    public static void checkNotNullList(List<?> objs) {
65      if (objs == null) {
66        throw new NullPointerException();
67      }
68      for (var obj : objs) {
69        if (obj == null) {
70          throw new NullPointerException();
71        }
72      }
73    }
74  
75    private static boolean isLetter(char ch) {
76      return ch >= 'A' && ch <= 'Z' || ch >= 'a' && ch <= 'z';
77    }
78  
79    private static boolean isDigit(char ch) {
80      return ch >= '0' && ch <= '9';
81    }
82  
83    private static boolean isIdentifierStart(char ch) {
84      return isLetter(ch) || ch == '_';
85    }
86  
87    private static boolean isIdentifierPart(char ch) {
88      return isLetter(ch) || isDigit(ch) || ch == '_';
89    }
90  
91    /**
92     * Returns whether {@code str} is a valid identifier.
93     *
94     * @param str the string to check
95     * @return whether {@code str} is a valid identifier
96     * @throws NullPointerException if {@code str} is {@code null}
97     */
98    public static boolean isIdentifier(String str) {
99      checkNotNull(str);
100     if (str.length() < 1 || !isIdentifierStart(str.charAt(0))) {
101       return false;
102     }
103     for (int i = 1; i < str.length(); i++) {
104       if (!isIdentifierPart(str.charAt(i))) {
105         return false;
106       }
107     }
108     return true;
109   }
110 
111   /**
112    * Checks that all strings in {@code strs} are valid identifiers.
113    *
114    * @param strs the strings to check
115    * @throws NullPointerException if {@code strs} or any string in {@code strs} is {@code null}
116    * @throws IllegalArgumentException if any string in {@code strs} is not a valid identifier
117    */
118   public static void checkIdentifier(String... strs) {
119     checkNotNull((Object[]) strs);
120     for (var str : strs) {
121       if (!isIdentifier(str)) {
122         throw new IllegalArgumentException(String.format("\"%s\" is not a valid identifier", str));
123       }
124     }
125   }

127   private static InputStream getResourceAsStream(String name) throws FileNotFoundException {
128     var in = Utils.class.getResourceAsStream(name);
129     if (in == null) {
130       throw new FileNotFoundException(String.format("%s (No such file or directory)", name));
131     }
132     return in;
133   }
134 
135   private static InputStream getLangSchemaJsonAsStream() throws FileNotFoundException {
136     return getResourceAsStream(LANG_SCHEMA_JSON);
137   }
138 
139   private static JsonObject getLangSchemaJsonAsJsonObject() {
140     try (var in = getLangSchemaJsonAsStream()) {
141       var jsonReaderFactory = Json.createReaderFactory(null);
142       try (var jsonReader = jsonReaderFactory.createReader(in, StandardCharsets.UTF_8)) {
143         return jsonReader.readObject();
144       } catch (JsonException e) {
145         throw new RuntimeException("Failed to read schema as JSON", e);
146       }
147     } catch (IOException e) {
148       throw new RuntimeException("Failed to read schema", e);
149     }
150   }
151 
152   private static JsonSchema getLangSchemaJsonAsJsonSchema() {
153     try (var in = getLangSchemaJsonAsStream()) {
154       return JSON_VALIDATION_SERVICE.readSchema(in, StandardCharsets.UTF_8);
155     } catch (IOException e) {
156       throw new RuntimeException("Failed to read schema", e);
157     } catch (JsonException e) {
158       throw new RuntimeException("Failed to read schema as JSON", e);
159     }
160   }
161 
162   private static JsonObject getJsonObject(JsonObject object, String name) {
163     if (!object.containsKey(name)) {
164       throw new RuntimeException("Invalid schema");
165     }
166     try {
167       return object.getJsonObject(name);
168     } catch (ClassCastException e) {
169       throw new RuntimeException("Invalid schema", e);
170     }
171   }
172 
173   private static JsonArray getJsonArray(JsonObject object, String name) {
174     if (!object.containsKey(name)) {
175       throw new RuntimeException("Invalid schema");
176     }
177     try {
178       return object.getJsonArray(name);
179     } catch (ClassCastException e) {
180       throw new RuntimeException("Invalid schema", e);
181     }
182   }
183 
184   private static String getArraySingleton(JsonArray array) {
185     if (array.size() != 1) {
186       throw new RuntimeException("Invalid schema");
187     }
188     try {
189       return array.getString(0);
190     } catch (ClassCastException e) {
191       throw new RuntimeException("Invalid schema", e);
192     }
193   }
194 
195   /**
196    * Returns the format version from the schema.
197    *
198    * @return the format version from the schema
199    */
200   public static String getFormatVersion() {
201     var langSchemaJson = getLangSchemaJsonAsJsonObject();
202     return getArraySingleton(
203         getJsonArray(
204             getJsonObject(getJsonObject(langSchemaJson, "properties"), "formatVersion"), "enum"));
205   }
206 
207   /**
208    * Returns whether {@code jsonLang} conforms to the schema.
209    *
210    * @param jsonLang a JSON representation of a MAL language
211    * @return whether {@code jsonLang} conforms to the schema
212    * @throws NullPointerException if {@code jsonLang} is {@code null}
213    */
214   public static boolean isValidJsonLang(JsonObject jsonLang) {
215     checkNotNull(jsonLang);
216     var jsonSchema = getLangSchemaJsonAsJsonSchema();
217     var jsonParserFactory = Json.createParserFactory(null);
218     try (var jsonParser = jsonParserFactory.createParser(jsonLang)) {
219       try (var jsonReader =
220           JSON_VALIDATION_SERVICE.createReader(jsonParser, jsonSchema, ProblemHandler.throwing())) {
221         jsonReader.readObject();
222         return true;
223       } catch (JsonValidatingException e) {
224         return false;
225       } catch (JsonException e) {
226         throw new RuntimeException("Failed to create JsonReader", e);
227       }
228     } catch (JsonException e) {
229       throw new RuntimeException("Failed to create JsonParser", e);
230     }
231   }
