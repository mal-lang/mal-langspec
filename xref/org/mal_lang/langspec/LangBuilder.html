17  package org.mal_lang.langspec;

19  import static org.mal_lang.langspec.Utils.checkIdentifier;
20  import static org.mal_lang.langspec.Utils.checkNotNull;
21  
22  import jakarta.json.JsonObject;
23  import java.util.ArrayList;
24  import java.util.LinkedHashMap;
25  import java.util.List;
26  import java.util.Map;

28  /** A builder for creating {@link Lang} objects. */
29  public final class LangBuilder {
30    private final Map<String, String> defines = new LinkedHashMap<>();
31    private final Map<String, CategoryBuilder> categories = new LinkedHashMap<>();
32    private final Map<String, AssetBuilder> assets = new LinkedHashMap<>();
33    private final List<AssociationBuilder> associations = new ArrayList<>();
34    private String license = null;
35    private String notice = null;
36  
37    LangBuilder() {}
38  
39    /**
40     * Adds a define to this {@code LangBuilder} object.
41     *
42     * @param key the key of the define
43     * @param value the value of the define
44     * @return this {@code LangBuilder} object
45     * @throws NullPointerException if {@code key} or {@code value} is {@code null}
46     * @throws IllegalArgumentException if {@code key} is not a valid identifier
47     */
48    public LangBuilder addDefine(String key, String value) {
49      checkIdentifier(key);
50      checkNotNull(value);
51      defines.put(key, value);
52      return this;
53    }
54  
55    /**
56     * Adds a category to this {@code LangBuilder} object.
57     *
58     * @param category the category to add
59     * @return this {@code LangBuilder} object
60     * @throws NullPointerException if {@code category} is {@code null}
61     */
62    public LangBuilder addCategory(CategoryBuilder category) {
63      checkNotNull(category);
64      categories.put(category.getName(), category);
65      return this;
66    }
67  
68    /**
69     * Adds an asset to this {@code LangBuilder} object.
70     *
71     * @param asset the asset to add
72     * @return this {@code LangBuilder} object
73     * @throws NullPointerException if {@code asset} is {@code null}
74     */
75    public LangBuilder addAsset(AssetBuilder asset) {
76      checkNotNull(asset);
77      assets.put(asset.getName(), asset);
78      return this;
79    }
80  
81    /**
82     * Adds an association to this {@code LangBuilder} object.
83     *
84     * @param association the association to add
85     * @return this {@code LangBuilder} object
86     * @throws NullPointerException if {@code association} is {@code null}
87     */
88    public LangBuilder addAssociation(AssociationBuilder association) {
89      checkNotNull(association);
90      associations.add(association);
91      return this;
92    }
93  
94    /**
95     * Returns the license of this {@code LangBuilder} object, or {@code null}.
96     *
97     * @return the license of this {@code LangBuilder} object, or {@code null}.
98     */
99    public String getLicense() {
100     return license;
101   }
102 
103   /**
104    * Sets the license of this {@code LangBuilder} object.
105    *
106    * @param license the license to set, or {@code null}
107    * @return this {@code LangBuilder} object
108    */
109   public LangBuilder setLicense(String license) {
110     this.license = license;
111     return this;
112   }
113 
114   /**
115    * Returns the notice of this {@code LangBuilder} object, or {@code null}.
116    *
117    * @return the notice of this {@code LangBuilder} object, or {@code null}.
118    */
119   public String getNotice() {
120     return notice;
121   }
122 
123   /**
124    * Sets the notice of this {@code LangBuilder} object.
125    *
126    * @param notice the notice to set, or {@code null}
127    * @return this {@code LangBuilder} object
128    */
129   public LangBuilder setNotice(String notice) {
130     this.notice = notice;
131     return this;
132   }
133 
134   /**
135    * Creates a new {@link Lang} object.
136    *
137    * @return a new {@link Lang} object
138    */
139   public Lang build() {
140     // Categories
141     var langCategories = new LinkedHashMap<String, Category>();
142     for (var category : categories.values()) {
143       var langCategory = category.build();
144       langCategories.put(langCategory.getName(), langCategory);
145     }
146 
147     // Assets
148     var langAssets = new LinkedHashMap<String, Asset>();
149     for (var asset : assets.values()) {
150       var langAsset = asset.build(langCategories);
151       langAssets.put(langAsset.getName(), langAsset);
152     }
153     for (var asset : assets.values()) {
154       var langAsset = langAssets.get(asset.getName());
155       asset.connectSuperAsset(langAssets);
156       asset.buildVariables(langAsset);
157       asset.buildAttackSteps(langAsset);
158     }
159 
160     // Associations
161     var langAssociations = new ArrayList<Association>();
162     for (var association : associations) {
163       var langAssociation = association.build(langAssets);
164       langAssociations.add(langAssociation);
165     }
166 
167     return new Lang(defines, langCategories, langAssets, langAssociations, license, notice);
168   }
169 
170   static LangBuilder fromJson(JsonObject jsonLang, Map<String, byte[]> icons) {
171     var langBuilder = Lang.builder();
172 
173     // Defines
174     var jsonDefines = jsonLang.getJsonObject("defines");
175     for (var key : jsonDefines.keySet()) {
176       langBuilder.addDefine(key, jsonDefines.getString(key));
177     }
178 
179     // Categories
180     var jsonCategories = jsonLang.getJsonArray("categories");
181     for (int i = 0; i < jsonCategories.size(); i++) {
182       var jsonCategory = jsonCategories.getJsonObject(i);
183       var categoryBuilder = CategoryBuilder.fromJson(jsonCategory);
184       langBuilder.addCategory(categoryBuilder);
185     }
186 
187     // Assets
188     var jsonAssets = jsonLang.getJsonArray("assets");
189     for (int i = 0; i < jsonAssets.size(); i++) {
190       var jsonAsset = jsonAssets.getJsonObject(i);
191       var assetBuilder = AssetBuilder.fromJson(jsonAsset);
192       langBuilder.addAsset(assetBuilder);
193       // SVG icon
194       var svgIconName = String.format("%s.svg", assetBuilder.getName());
195       if (icons.containsKey(svgIconName)) {
196         assetBuilder.setSvgIcon(icons.get(svgIconName));
197       }
198       // PNG icon
199       var pngIconName = String.format("%s.png", assetBuilder.getName());
200       if (icons.containsKey(pngIconName)) {
201         assetBuilder.setPngIcon(icons.get(pngIconName));
202       }
203     }
204 
205     // Associations
206     var jsonAssociations = jsonLang.getJsonArray("associations");
207     for (int i = 0; i < jsonAssociations.size(); i++) {
208       var jsonAssociation = jsonAssociations.getJsonObject(i);
209       var associationBuilder = AssociationBuilder.fromJson(jsonAssociation);
210       langBuilder.addAssociation(associationBuilder);
211     }
212 
213     return langBuilder;
214   }
